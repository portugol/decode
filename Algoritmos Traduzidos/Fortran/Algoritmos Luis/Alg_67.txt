program Alg_67

real :: valor
real :: taxa
real :: tempo
real :: prest

valor=0.0
taxa=0.0
tempo=0.0

print *,"introduza o valor da prestação"
read *,valor
print *,"introduza a taxa"
read *,taxa
print *,"introduza o numero de meses em atraso"
read *,tempo

prest=valor+(valor*(taxa/100)*tempo)
print *,"a prestação e, atraso é :",prest
end
