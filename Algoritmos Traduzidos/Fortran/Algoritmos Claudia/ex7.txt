program divisivel
integer :: numero
numero=0

print *,"Digite um numero"
read *,numero

if(mod(numero,10)==0)then
  print *,"é multiplo de 10"
  else if(mod(numero,2)==0)then
    print *,"é multiplo de 2"
    else if(mod(numero,5)==0)then
      print *,"é multiplo de 5"
      else
        print *,"nao é multiplo de 2 nem de 5"
endif
end
