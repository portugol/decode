FUNÇÕES

Crie um algoritmo que permita o cáculo de um factorial.

package funcao;

public class Funcao {

    public static void main(String[] args) {
      int l=5;
     
      int j;
      j=fact(l);
      System.out.println(j);
    }
    
    public static int fact(int k){
        if (k>2) {
            return k*fact(k-1);
        }else{
           return k; 
        }
        
    }
}
