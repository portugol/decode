import java.util.Scanner;

public class Ex4 {

    public static void main(String[] args) {
        int n;
        Scanner scanner = new Scanner(System.in);

        System.out.println("Digite um numero");
        n = scanner.nextInt();
        if (n % 2 == 0) {
            System.out.println("Numero par");
        } else {
            System.out.println("Numero ímpar");
        }
    }
}
