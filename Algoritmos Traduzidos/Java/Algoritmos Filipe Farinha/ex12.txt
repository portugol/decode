import java.util.Scanner;

public class Ex12 {

    public static void main(String[] args) {
        int num, soma, produto;
        soma = 0;
        produto = 1;
        Scanner scanner = new Scanner(System.in);
        System.out.println("introduz um nº :");
        num = scanner.nextInt();

        for (int i = 1; i <= num; i++) {
            soma += i;
            produto *= i;
        }
        System.out.println("soma =" + soma + "\n" + "produto =" + produto);
    }
}
