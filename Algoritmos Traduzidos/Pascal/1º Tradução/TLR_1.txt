Program SalarioPRG;
uses crt;
var salario : Real;
    prof : String;
Begin
     readln(salario);
     readln(prof);
     If prof = 'tecnico' then
        BEGIN
          salario := salario * 1.05;
        END
     Else
         Begin
              IF prof = 'gerente' then
                 BEGIN
                      salario := salario * 1.03;
                 END
              Else
                  BEGIN
                       salario := salario * 1.01;
                  END
         End;
writeln('O novo Salario ‚ ' , salario);
READLN;
End.
