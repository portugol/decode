    Private Sub Button4_Click(sender As System.Object, e As System.EventArgs) Handles Button4.Click
        Dim P As Integer
        P = 0
        Dim num1 As Integer
        num1 = 0
        Dim num As Integer
        num1 = 1
        num = InputBox("DIGITE UM NUMERO:")
        P = num1 * (num1 + 1) * (num1 + 2)
        While (P < num)
            num1 = num1 + 1
            P = num1 * (num1 + 1) * (num1 + 2)
        End While
        If P = num Then
            MsgBox("TRIANGULAR")
        Else
            MsgBox("NÃO TRIANGULAR")
        End If
    End Sub
