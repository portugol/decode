Private Sub Button1_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles Button1.Click
        Dim mes As Integer
        mes = 0
        While (mes < 1) Or (mes > 12)
            mes = InputBox("Introduza um mês:")
        End While
        MsgBox("Mes introduzido: " + mes.ToString)
End Sub
