{
    int n[3][3];
    int l;
    int c;
    int t;
    l=0;
    c=0;
    t=0;
    for(l=0; l<=2; l++)
    {
        for(c=0; c<=2; c++)
        {
            cout<<"\nIntroduza o elemento linha "<<l+1<<" coluna "<<c+1<<" :";
            cin>>n[l][c];
        }
    }
    cout<<"\n Diagonal Principal \n";
    for(l=0; l<=2; l++)
    {
        cout<< n[l][l]<<"\n";
    }
    for(t=0; t<=l; t++)
    {
        cout<<"\t";
    }
}
