#include <iostream>
using namespace std;

int main()
{
    float percurso, consumo;
    char tipo;
    percurso=0;
    consumo=0;
    cout<<"Seleccione o tipo de carro: (A/B/C) ";
    cin>>tipo;
    cout<<"Insira o numero de km que deseja efectuar ";
    cin>>percurso;
    if(tipo=='C' || tipo=='c')
    {
        consumo=percurso/12;
    }
    else
    {
        if(tipo=='B' || tipo=='b')
        {
            consumo=percurso/10;
        }
        else
        {
            if(tipo=='A' || tipo=='a')
            {
                consumo=percurso/8;
            }
            else
            {
                consumo=0;
            }
        }
    }
    if (consumo!=0)
    {
        cout<<"Consumo estimado em litros: "<<consumo;
    }
    else
    {
        cout<<"Modelo Inexistente";
    }
}
