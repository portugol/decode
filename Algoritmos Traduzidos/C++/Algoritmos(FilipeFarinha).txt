{
    int imc;
    cout<<"Digite o valor do IMC (Índice de Massa Corporal): ";
    cin>>imc;
    if ( imc < 20 )
    {
        cout<<"Abaixo do peso";
    }
    else
    {
        if (imc >= 20 && imc <= 24)
        {
            cout<<"Peso ideal";
        }
        else
        {
            if (imc >= 25 && imc <= 39)
            {
                cout<<"Excesso de peso";
            }
            else
            {
                if (imc >= 30 && imc <= 39)
                {
                    cout<<"Obesidade";
                }
                else
                {
                    if ( imc > 39 )
                    {
                        cout<<"Obesidade mórbida";
                    }
                    else
                    {
                        cout<<"Valor inválido";
                    }
                }
            }
        }
    }
}

#include <iostream>
using namespace std;

int main()
{
int li=0, ls=0, s=0;
	cout<<"Limite inferior: ";
	cin>>li;
	cout<<"Limite superior: ";
	cin>>ls;
	while (li <= ls)
	{
		s = s + li;
		li = li + 1;
	}
	cout<<s;

}
