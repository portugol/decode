    for(k=0; k<=n; k++)
    {
        if (k%2==0)
        {
            pi=pi+1/(2*k+1);
        }
        else
        {
            pi=pi-1/(2*k+1);
        }
    }
    pi=4*pi;
    printf("O valor de pi e %f",pi);

#include <stdio.h>
#include <stdlib.h>

void main()
{
    int n[3][3];
    int l;
    int c;
    int t;
    l=0;
    c=0;
    t=0;
    for(l=0; l<=2; l++)
    {
        for(c=0; c<=2; c++)
        {
            printf("\nIntroduza o elemento linha %d coluna %d :",l+1,c+1);
            scanf("%d",&n[l][c]);
        }
    }
    printf("\n Diagonal Principal \n");
    for(l=0; l<=2; l++)
    {
        printf("%d \n", n[l][l]);
    }
    for(t=0; t<=l; t++)
    {
        printf("\t");
    }
}
