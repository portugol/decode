#include <stdlib.h>

void main()
{
    int opcao;
    printf("1-Verao \n");
    printf("2-Outono \n");
    printf("3-Inverno \n");
    printf("4-Primavera \n");
    while(opcao<1 || opcao>4)
    {
        printf("Escolha a opcao do menu referente ao periodo desejado:");
        scanf("%d",&opcao);
    }
    switch(opcao)
    {
    case 1:
        printf("Verao");
        break;
    case 2:
        printf("Outono");
        break;
    case 3:
        printf("Inverno");
        break;
    case 4:
        printf("Primavera");
        break;
    default:
        break;
    }
}

{

    printf("DIGITE UM NUMERO:");
    scanf("%d",&NUM);
    P=NUM1*(NUM1+1)*(NUM1+2);
    while(P<NUM)
    {
        NUM1=NUM1+1;

#include <stdlib.h>

void main()
{
    int k;
    int n;
    float pi;
    printf("Digite o valor de n ");
    scanf("%d",&n);
    for(k=0; k<=n; k++)
    {
        if (k%2==0)
        {
            pi=pi+1/(2*k+1);
        }
        else
        {
            pi=pi-1/(2*k+1);
        }
    }
    pi=4*pi;
    printf("O valor de pi e %f",pi);
}
