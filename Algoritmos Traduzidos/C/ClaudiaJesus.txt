#include <stdio.h>
#include <stdlib.h>

void main()
{
    float percurso, consumo;
    char tipo;
    percurso=0;
    consumo=0;
    printf("Seleccione o tipo de carro: (A/B/C) ");
    scanf("%c",&tipo);
    printf("Insira o numero de km que deseja efectuar ");
    scanf("%f",&percurso);
    if(tipo=='C' || tipo=='c')
    {
        consumo=percurso/12;
    }
    else
    {
        if(tipo=='B' || tipo=='b')
        {
            consumo=percurso/10;
        }
        else
        {
            if(tipo=='A' || tipo=='a')
            {
                consumo=percurso/8;
            }
            else
            {
                consumo=0;
            }
        }
    }
    if (consumo!=0)
    {
        printf("Consumo estimado em litros: %f",consumo);

    }
    else
    {
        printf("Modelo Inexistente");
    }
}
