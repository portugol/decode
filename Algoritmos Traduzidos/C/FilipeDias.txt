#include <stdio.h>
#include <stdlib.h>

void main()
{
    int n;
    int num;
    int fat;
    n=0;
    num=0;
    fat=0;
    printf("\nEste programa calcula N factorial");
    printf("\nDigite o valor para ser calculado:");
    scanf("%d",&num);
    if(num>=0)
    {
        fat=1;
        n=num;
        while(n>0)
        {
            fat=fat*n;
            n=n-1;
        }
        printf("O numero factorial de %d e %d",num,fat);
    }
    else
    {
        printf("O calculo do fact aplica-se somente aos numeros naturais");
    }
    printf("\n");
}
