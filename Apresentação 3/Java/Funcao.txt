package funcao;

public class Funcao {

    public static void main(String[] args) {
      int l=5;
     
      int j;
      j=fact(l);
      System.out.println(j);
    }
    
    public static int fact(int k){
        if (k>2) {
            k=k*fact(k-1);
        }else{
           k=k; 
        }
        return k;
    }
}
